programa {
  funcao inicio() {

    // Variáveis
    real mensalidade, transportedia, alimentacaodia, customensal, custoanual

    // Entradas
    escreva ("Digite o valor da mensalidade:")
    leia(mensalidade)
    escreva ("Digite o seu gasto diário com transporte:")
    leia (transportedia)
    escreva ("Digite seu gasto diário com alimentação:")
    leia (alimentacaodia)

    // Processamento
    customensal = ( mensalidade + transportedia * 16 + alimentacaodia * 16)  // Aulas presenciais de segunda a quinta, portanto, 16 dias no mês
    custoanual = (customensal * 12)

    // Saída
    escreva ("Seu gasto mensal com a Faculdade é:")
    escreva (customensal)
    escreva ("\n")
    escreva ("Seu custo anual com a Faculdade é:")
    escreva (custoanual)
    
  }
}
